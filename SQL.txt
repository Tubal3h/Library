-- =========================
-- TABELLE DI RIFERIMENTO
-- =========================

CREATE TABLE books_names(
    book_name_id SERIAL PRIMARY KEY,
    title TEXT UNIQUE
);

CREATE TABLE category(
    category_id SERIAL PRIMARY KEY,
    category_name TEXT UNIQUE
);

CREATE TABLE author(
    author_id SERIAL PRIMARY KEY,
    author_name VARCHAR(32),
    author_last_name VARCHAR(32)
);

CREATE TABLE publisher(
    publisher_id SERIAL PRIMARY KEY,
    publisher_name VARCHAR(32)
);

-- Edizione: titolo + autore + publisher + data + ISBN
CREATE TABLE edition(
    edition_id SERIAL PRIMARY KEY,
    book_name_id INTEGER NOT NULL,
    author_id INTEGER NOT NULL,
    publisher_id INTEGER NOT NULL,
    publishing_date DATE,
    isbn VARCHAR(13) UNIQUE,
    FOREIGN KEY(book_name_id) REFERENCES books_names(book_name_id) ON DELETE CASCADE,
    FOREIGN KEY(author_id) REFERENCES author(author_id) ON DELETE CASCADE,
    FOREIGN KEY(publisher_id) REFERENCES publisher(publisher_id) ON DELETE CASCADE
);

-- Copia fisica del libro
CREATE TABLE books(
    book_id SERIAL PRIMARY KEY,
    edition_id INTEGER NOT NULL,
    category_id INTEGER NOT NULL,
    status TEXT NOT NULL CHECK(status IN ('disponibilita', 'in prestito')),
    FOREIGN KEY(edition_id) REFERENCES edition(edition_id) ON DELETE CASCADE,
    FOREIGN KEY(category_id) REFERENCES category(category_id) ON DELETE CASCADE
);

-- Utenti
CREATE TABLE users(
    users_id SERIAL PRIMARY KEY,
    user_name VARCHAR(32),
    user_last_name VARCHAR(32),
    email VARCHAR(64) UNIQUE,
    pass VARCHAR(32),
    roles TEXT NOT NULL CHECK(roles IN ('role_user', 'role_admin'))
);

-- Noleggi
CREATE TABLE rental_record(
    rental_id SERIAL PRIMARY KEY,
    book_id INTEGER NOT NULL,
    users_id INTEGER NOT NULL,
    rental_date TIMESTAMP,
    rental_expired DATE,
    rental_ended TIMESTAMP NULL,
    FOREIGN KEY(book_id) REFERENCES books(book_id) ON DELETE CASCADE,
    FOREIGN KEY(users_id) REFERENCES users(users_id) ON DELETE CASCADE
);
